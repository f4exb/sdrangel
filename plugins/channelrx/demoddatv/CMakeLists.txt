project(datv)

set(datv_SOURCES
	datvdemod.cpp
	datvdemodgui.cpp
	datvdemodplugin.cpp
	datvideostream.cpp
	datvideorender.cpp
)

set(datv_HEADERS
	datvdemod.h
	datvdemodgui.h
	datvdemodplugin.h
	datvideostream.h
	datvideorender.h
)

	datvdemodgui.ui
)

	.
	${CMAKE_CURRENT_BINARY_DIR}
	${AVCODEC_INCLUDE_DIRS}
	${AVFORMAT_INCLUDE_DIRS}
	${SWSCALE_INCLUDE_DIRS}
)

	${datv_SOURCES}
	${datv_HEADERS_MOC}
	${datv_FORMS_HEADERS}
)

	${QT_LIBRARIES}
	sdrbase
	sdrgui
	${AVCODEC_LIBRARIES}
	${AVFORMAT_LIBRARIES}
	${SWSCALE_LIBRARIES}
)
